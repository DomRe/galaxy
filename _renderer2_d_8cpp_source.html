323  void Renderer2D::init(std::string_view layers)
324  {
325  const auto opt = json::parse_from_disk(layers);
326  if (opt != std::nullopt)
327  {
328  const auto& root = opt.value().at("layers");
329  for (const auto& [name, layer] : root.items())
330  {
331  m_layer_data.emplace(name, layer.get<int>());
332  }
333 
334  // Set pointers to layer data.
335  for (auto& [name, layer] : m_layer_data)
336  {
337  for (const auto& [key, atlas] : SL_HANDLE.texturebook()->get_all())
338  {
339  layer.m_batches.try_emplace(key);
340  layer.m_batches[key].add_texture(atlas.gl_texture());
341  }
342 
343  m_layers.push_back(&layer);
344  }
345 
346  // Make sure order is correct.
347  std::sort(std::execution::par, m_layers.begin(), m_layers.end(), [&](auto* left, auto* right) {
348  return left->get_layer() < right->get_layer();
349  });
350  }
351  else
352  {
353  GALAXY_LOG(GALAXY_FATAL, "Failed to create renderlayers from path: {0}.", layers);
354  }
355  }
